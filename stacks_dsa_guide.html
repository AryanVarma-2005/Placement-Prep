Stack Fundamentals
> before any algorithm, understand what a stack IS and why it exists
BASICWhat Is a Stack?

A stack is a LIFO (Last In, First Out) data structure. Think of a stack of plates — you always add to the top and remove from the top. The last plate placed is the first one taken off.

> STACK STATE AFTER PUSHING 10, 25, 7, 42:

▼ PUSH direction ▼
42idx: 3
7idx: 2
25idx: 1
10idx: 0
▓ BOTTOM ▓
⚡

The core rule: You can ONLY interact with the TOP of the stack. You cannot access elements in the middle or bottom without first popping off everything above them. This restriction is exactly what makes stacks useful — it forces a disciplined ordering.

> WHERE STACKS APPEAR IN THE REAL WORLD:

  • Function call stack — your CPU uses a stack for function calls. Each function call is "pushed", and when it returns, it's "popped".
  • Undo/Redo — Ctrl+Z in any editor pops from an undo stack.
  • Browser back button — visited pages are a stack; "back" = pop.
  • Parenthesis matching — compilers use stacks to validate brackets.
  • Expression evaluation — calculators convert infix to postfix using stacks.
  • DFS (Depth-First Search) — explicit stack or recursion stack.
OperationDescriptionTimeSpace
push(x)Add x to top of stackO(1)O(1)
pop()Remove & return top elementO(1)O(1)
peek() / top()View top element without removingO(1)O(1)
isEmpty()Check if stack has no elementsO(1)O(1)

BASICPython List as a Stack

Python's built-in list works perfectly as a stack. append() is push, pop() is pop, [-1] is peek. All O(1) amortized.

stack_list.py
# ══════════════════════════════════════════════════════
#  PYTHON LIST AS STACK — the standard approach in DSA
# ══════════════════════════════════════════════════════

stack = []

# ── PUSH: append to end ──────────────────────────────
stack.append(10)       # stack: [10]
stack.append(25)       # stack: [10, 25]
stack.append(7)        # stack: [10, 25, 7]
stack.append(42)       # stack: [10, 25, 7, 42]

# ── PEEK: view top without removing ──────────────────
top = stack[-1]         # 42 — NEVER pops
print(f"Top: {top}")     # Top: 42

# ── POP: remove and return top ────────────────────────
val = stack.pop()        # returns 42, stack: [10, 25, 7]
print(f"Popped: {val}") # Popped: 42

# ── IS EMPTY ──────────────────────────────────────────
print(not stack)         # False (stack has items)
print(len(stack) == 0)  # False

# ── SIZE ──────────────────────────────────────────────
print(len(stack))        # 3

# ── SAFE POP (avoid IndexError on empty stack) ────────
def safe_pop(stack):
    return stack.pop() if stack else None

# ── SAFE PEEK ─────────────────────────────────────────
def peek(stack):
    return stack[-1] if stack else None

# ── ITERATE (does NOT modify stack) ──────────────────
for item in reversed(stack):  # top to bottom
    print(item)

# ── CLEAR THE STACK ───────────────────────────────────
stack.clear()
# or: stack = []
⚠

Common mistake: Calling stack.pop(0) removes from the FRONT (bottom of stack) — that's a QUEUE, not a stack! Always use stack.pop() with no argument for stack behavior.

BASICStack Class — Clean OOP
stack_class.py
# ── Stack Class (list-backed) ────────────────────────
class Stack:
    def __init__(self):
        self._data = []

    def push(self, val):
        self._data.append(val)         # O(1) amortized

    def pop(self):
        if self.is_empty():
            raise IndexError("Stack underflow")
        return self._data.pop()        # O(1) amortized

    def peek(self):
        if self.is_empty():
            return None
        return self._data[-1]          # O(1)

    def is_empty(self): return len(self._data) == 0
    def size(self):     return len(self._data)
    def __len__(self):  return len(self._data)
    def __str__(self):  return f"Stack(top→{self._data[::-1]})"

# ── Stack backed by Linked List (O(1) guaranteed, no resize) ─
class Node:
    def __init__(self, val, next=None): self.val=val; self.next=next

class LinkedStack:
    def __init__(self): self.head=None; self._size=0

    def push(self, val):
        self.head = Node(val, self.head)   # new node → old head
        self._size += 1

    def pop(self):
        if not self.head: raise IndexError("Empty stack")
        val = self.head.val
        self.head = self.head.next         # head moves down
        self._size -= 1
        return val

    def peek(self): return self.head.val if self.head else None
    def is_empty(self): return self.head is None

# ── deque: the FASTEST stack in Python ───────────────
from collections import deque

stack = deque()
stack.append(10)        # push — O(1) GUARANTEED (not amortized)
stack.append(20)
val = stack.pop()        # pop  — O(1) GUARANTEED
top = stack[-1]          # peek — O(1)

# Use deque when performance matters.
# Use list for simplicity in interview solutions.
MEDIUMMin Stack — O(1) Get Minimum

Design a stack that supports push, pop, peek, and getMin() — all in O(1). The trick: maintain a parallel min-tracking stack.

📡

Core insight: Each slot in the min-stack stores "the minimum of all elements at-or-below this point". When you push, record min(new_val, current_min). When you pop, the min-stack automatically reflects the correct minimum for remaining elements.

min_stack.py
class MinStack:
    def __init__(self):
        self.stack     = []   # main stack
        self.min_stack = []   # parallel: tracks current min

    def push(self, val):
        self.stack.append(val)
        # min_stack top = min so far (including new val)
        current_min = min(val, self.min_stack[-1] if self.min_stack else val)
        self.min_stack.append(current_min)

    def pop(self):
        self.min_stack.pop()    # always pop both together
        return self.stack.pop()

    def top(self):
        return self.stack[-1]

    def get_min(self):
        return self.min_stack[-1]    # O(1)!

# Demo
ms = MinStack()
ms.push(-2);  ms.push(0);  ms.push(-3)
print(ms.get_min())  # -3  (minimum in stack)
ms.pop()
print(ms.top())      # 0   (top after pop)
print(ms.get_min())  # -2  (new minimum after -3 is gone)

# ── Variant: MaxStack — same idea, track max instead ─
class MaxStack:
    def __init__(self): self.stack=[]; self.max_s=[]
    def push(self,v):
        self.stack.append(v)
        self.max_s.append(max(v, self.max_s[-1] if self.max_s else v))
    def pop(self): self.max_s.pop(); return self.stack.pop()
    def get_max(self): return self.max_s[-1]

BASICValid Parentheses

Given a string of brackets, determine if it's valid. Push open brackets; on close brackets, check the top matches. Classic stack application.

valid_parens.py
def is_valid(s):
    stack = []
    mapping = {')': '(', '}': '{', ']': '['}

    for ch in s:
        if ch in mapping:              # closing bracket
            top = stack.pop() if stack else '#'
            if mapping[ch] != top:     # mismatch!
                return False
        else:
            stack.append(ch)            # opening bracket

    return not stack  # valid only if stack is empty at end

print(is_valid("()[]{}"))   # True
print(is_valid("([)]"))     # False
print(is_valid("{[]}"))     # True
print(is_valid("("))        # False (unclosed)

# ── Minimum additions to make valid ─────────────────
def min_add_to_valid(s):
    open_count = close_count = 0
    for ch in s:
        if ch == '(':
            open_count += 1
        elif open_count > 0:
            open_count -= 1    # matched a (
        else:
            close_count += 1   # unmatched )
    return open_count + close_count

print(min_add_to_valid("())"))    # 1
print(min_add_to_valid("((("))    # 3
MEDIUMEvaluate Postfix / RPN

Reverse Polish Notation (Postfix): operators come after operands. 3 4 + = 7. Push numbers; on operator, pop two, compute, push result.

rpn_eval.py
def eval_rpn(tokens):
    stack = []
    ops = {
        '+': lambda a,b: a+b,
        '-': lambda a,b: a-b,
        '*': lambda a,b: a*b,
        '/': lambda a,b: int(a/b)   # truncate toward zero
    }
    for t in tokens:
        if t in ops:
            b = stack.pop()     # second operand (popped first!)
            a = stack.pop()     # first operand
            stack.append(ops[t](a, b))
        else:
            stack.append(int(t))

    return stack[0]   # final answer

print(eval_rpn(["2","1","+","3","*"]))       # 9  → (2+1)*3
print(eval_rpn(["4","13","5","/","+"]))       # 6  → 4+(13/5)
print(eval_rpn(["10","6","9","3","+","-11","*","/","*","17","+","5","+"]))
MEDIUMDecode String

Given k[encoded_string], return the decoded string. Example: 3[a2[bc]] → abcbcabcbcabcbc. Two stacks: one for counts, one for strings.

decode_string.py
def decode_string(s):
    count_stack  = []   # stack for repetition counts
    string_stack = []   # stack for partial strings
    current_str  = ""
    current_num  = 0

    for ch in s:
        if ch.isdigit():
            current_num = current_num * 10 + int(ch)  # handle multi-digit: "12[a]"

        elif ch == '[':
            count_stack.append(current_num)    # save count
            string_stack.append(current_str)   # save current string
            current_num = 0
            current_str = ""

        elif ch == ']':
            repeat = count_stack.pop()
            prev   = string_stack.pop()
            current_str = prev + current_str * repeat  # expand!

        else:
            current_str += ch

    return current_str

print(decode_string("3[a]2[bc]"))      # "aaabcbc"
print(decode_string("3[a2[c]]"))       # "accaccacc"
print(decode_string("2[abc]3[cd]ef"))  # "abcabccdcdcdef"
MEDIUMRemove Adjacent Duplicates & Asteroid Collision
duplicates_collision.py
# ── Remove Adjacent Duplicates — O(n) ───────────────
# "abbaca" → "ca" (remove pairs step by step)
def remove_duplicates(s):
    stack = []
    for ch in s:
        if stack and stack[-1] == ch:
            stack.pop()    # adjacent duplicate → remove both
        else:
            stack.append(ch)
    return "".join(stack)

print(remove_duplicates("abbaca"))   # "ca"
print(remove_duplicates("azxxzy"))   # "ay"

# ── Remove k adjacent duplicates ────────────────────
def remove_k_duplicates(s, k):
    # stack stores (char, count) pairs
    stack = []
    for ch in s:
        if stack and stack[-1][0] == ch:
            stack[-1][1] += 1
            if stack[-1][1] == k:
                stack.pop()         # k in a row → remove all k
        else:
            stack.append([ch, 1])
    return "".join(ch * cnt for ch, cnt in stack)

print(remove_k_duplicates("deeedbbcccbdaa", 3))  # "aa"

# ── Asteroid Collision ───────────────────────────────
# Positive = moves right, negative = moves left
# Collision only if right-moving meets left-moving
def asteroid_collision(asteroids):
    stack = []
    for a in asteroids:
        while stack and a < 0 < stack[-1]:
            if stack[-1] < -a:
                stack.pop()         # right asteroid destroyed
                continue
            elif stack[-1] == -a:
                stack.pop()         # both destroyed
            break                   # new asteroid destroyed
        else:
            stack.append(a)         # no collision
    return stack

print(asteroid_collision([5, 10, -5]))        # [5, 10]
print(asteroid_collision([8, -8]))             # []
print(asteroid_collision([10, 2, -5]))         # [10]
print(asteroid_collision([-2, -1, 1, 2]))      # [-2, -1, 1, 2]

Monotonic Stack
> the most powerful stack pattern. converts O(n²) problems to O(n).
PATTERNWhat Is a Monotonic Stack?

A monotonic stack is a stack where elements are always maintained in monotonically increasing or decreasing order. When a new element breaks the monotonic property, you pop elements until the property is restored.

⚡

When to use: Any problem asking for "next greater", "next smaller", "previous greater", "previous smaller" element — monotonic stack solves it in O(n). Without it, brute force is O(n²).

> MONOTONIC INCREASING STACK — elements stay in increasing order:

2
5
8
base
increasing
→

Push 3. 3 < 2 (top)? NO.

Pop 2 (2 < 3), pop 5? No wait—

Key: pop WHILE stack top < new element (for decreasing stack) or > new element (for increasing stack). Then push.

monotonic_stack.py
# ── Template: Monotonic DECREASING stack ─────────────
# Useful for: next greater element (NGE)
arr = [4, 5, 2, 10, 8]
n = len(arr)
result = [-1] * n        # -1 = no greater element
stack = []               # stores INDICES (not values)

for i in range(n):
    # Pop while current element > stack top element
    while stack and arr[i] > arr[stack[-1]]:
        idx = stack.pop()
        result[idx] = arr[i]     # arr[i] is NGE for arr[idx]
    stack.append(i)

# result[i] = next greater element to the right of arr[i]
print(result)   # [5, 10, 10, -1, -1]
# arr[0]=4 → NGE is 5
# arr[1]=5 → NGE is 10
# arr[2]=2 → NGE is 10
# arr[3]=10 → no greater → -1
# arr[4]=8  → no greater → -1

# ── Template: Monotonic INCREASING stack ─────────────
# Useful for: next smaller element
result2 = [-1] * n
stack2 = []

for i in range(n):
    while stack2 and arr[i] < arr[stack2[-1]]:  # flip to <
        idx = stack2.pop()
        result2[idx] = arr[i]
    stack2.append(i)

print(result2)  # [2, 2, -1, 8, -1]
PATTERNDaily Temperatures & Stock Span
temperatures_span.py
# ── Daily Temperatures — O(n) ───────────────────────
# For each day, how many days until a warmer temperature?
def daily_temperatures(temps):
    n = len(temps)
    result = [0] * n
    stack = []   # monotonic decreasing (by temperature)

    for i, t in enumerate(temps):
        while stack and temps[stack[-1]] < t:
            prev_day = stack.pop()
            result[prev_day] = i - prev_day   # days waited
        stack.append(i)

    return result  # 0 means no warmer day found

temps = [73,74,75,71,69,72,76,73]
print(daily_temperatures(temps))
# [1, 1, 4, 2, 1, 1, 0, 0]

# ── Stock Span Problem — O(n) ────────────────────────
# Span of day i = consecutive days before i with price ≤ today
def stock_span(prices):
    n = len(prices)
    span = [1] * n
    stack = []   # monotonic decreasing stack of indices

    for i in range(n):
        while stack and prices[stack[-1]] <= prices[i]:
            stack.pop()
        span[i] = i - stack[-1] if stack else i + 1
        stack.append(i)

    return span

prices = [100, 80, 60, 70, 60, 75, 85]
print(stock_span(prices))
# [1, 1, 1, 2, 1, 4, 6]
# day 5 (price=75): 4 days of consecutive ≤ prices (75,60,70,60)
# day 6 (price=85): 6 days (entire array so far)

# ── Previous Greater Element ─────────────────────────
def prev_greater(arr):
    n = len(arr)
    result = [-1] * n
    stack = []
    for i in range(n):
        while stack and stack[-1] <= arr[i]:
            stack.pop()
        result[i] = stack[-1] if stack else -1
        stack.append(arr[i])
    return result

print(prev_greater([4, 5, 2, 10, 8]))
# [-1, -1, 5, -1, 10]

HARDLargest Rectangle in Histogram

Find the largest rectangle in a histogram. For each bar, find how far left and right it can extend at its height. Monotonic increasing stack gives O(n).

🧠

Key insight: When we find a bar shorter than the stack top, the top bar's "territory" ends. We pop it and calculate the area it could have formed — its width extends from the new stack top's position to the current index.

histogram.py
def largest_rectangle(heights):
    stack = []   # monotonic increasing stack of indices
    max_area = 0
    n = len(heights)

    for i in range(n + 1):   # +1 to flush remaining bars
        h = heights[i] if i < n else 0

        while stack and h < heights[stack[-1]]:
            height = heights[stack.pop()]
            # Width: from stack top to current index
            width  = i if not stack else i - stack[-1] - 1
            max_area = max(max_area, height * width)

        stack.append(i)

    return max_area

print(largest_rectangle([2,1,5,6,2,3]))   # 10 (bars 5,6)
print(largest_rectangle([2,4]))              # 4

# ── Maximal Rectangle in Binary Matrix ──────────────
# Extension: treat each row as a histogram!
def maximal_rectangle(matrix):
    if not matrix or not matrix[0]: return 0
    n = len(matrix[0])
    heights = [0] * n
    max_area = 0

    for row in matrix:
        # Update running histogram height
        for j in range(n):
            heights[j] = heights[j] + 1 if row[j] == '1' else 0
        max_area = max(max_area, largest_rectangle(heights))

    return max_area
HARDSum of Subarray Minimums

Find the sum of min(subarray) for all subarrays. Key: for each element, find how many subarrays it is the minimum of using monotonic stack.

subarray_mins.py
def sum_subarray_mins(arr):
    MOD = 10**9 + 7
    n = len(arr)
    result = 0
    stack = []   # monotonic increasing — stores indices

    for i in range(n + 1):
        while stack and (i == n or arr[stack[-1]] >= arr[i]):
            mid = stack.pop()
            left  = stack[-1] if stack else -1
            right = i

            # arr[mid] is minimum for (mid-left) * (right-mid) subarrays
            count = (mid - left) * (right - mid)
            result = (result + arr[mid] * count) % MOD

        stack.append(i)

    return result

print(sum_subarray_mins([3,1,2,4]))
# 17
# Subarrays: [3]=3, [1]=1, [2]=2, [4]=4,
#            [3,1]=1, [1,2]=1, [2,4]=2,
#            [3,1,2]=1, [1,2,4]=1,
#            [3,1,2,4]=1   → sum = 17

# ── 132 Pattern ─────────────────────────────────────
# Find if there exists i < j < k: nums[i] < nums[k] < nums[j]
def find132pattern(nums):
    stack = []
    k = float('-inf')   # tracks nums[k] (the "2" in 132)

    for num in reversed(nums):   # scan right to left
        if num < k:
            return True       # found nums[i] < k
        while stack and stack[-1] < num:
            k = stack.pop()    # update k: largest value less than num
        stack.append(num)

    return False

print(find132pattern([1,2,3,4]))      # False
print(find132pattern([3,1,4,2]))      # True (1,4,2)
print(find132pattern([-1,3,2,0]))     # True (-1,3,0)

# ── Undo/Redo System ────────────────────────────────
class TextEditor:
    def __init__(self):
        self.text      = ""
        self.undo_stack = []   # stores previous states
        self.redo_stack = []   # stores undone states

    def type(self, chars):
        self.undo_stack.append(self.text)   # save current
        self.redo_stack.clear()              # new action clears redo
        self.text += chars

    def undo(self):
        if not self.undo_stack: return
        self.redo_stack.append(self.text)
        self.text = self.undo_stack.pop()

    def redo(self):
        if not self.redo_stack: return
        self.undo_stack.append(self.text)
        self.text = self.redo_stack.pop()

editor = TextEditor()
editor.type("Hello")
editor.type(" World")
print(editor.text)    # "Hello World"
editor.undo()
print(editor.text)    # "Hello"
editor.redo()
print(editor.text)    # "Hello World"

# ── Browser History ──────────────────────────────────
class BrowserHistory:
    def __init__(self, homepage):
        self.back_stack    = []          # pages we can go back to
        self.forward_stack = []          # pages we can go forward to
        self.current       = homepage

    def visit(self, url):
        self.back_stack.append(self.current)
        self.forward_stack.clear()       # visiting clears forward
        self.current = url

    def back(self, steps):
        for _ in range(steps):
            if not self.back_stack: break
            self.forward_stack.append(self.current)
            self.current = self.back_stack.pop()
        return self.current

    def forward(self, steps):
        for _ in range(steps):
            if not self.forward_stack: break
            self.back_stack.append(self.current)
            self.current = self.forward_stack.pop()
        return self.current

# ── Infix to Postfix (Shunting Yard Algorithm) ───────
def infix_to_postfix(expr):
    precedence = {'+':1, '-':1, '*':2, '/':2, '^':3}
    output = []
    ops    = []

    for token in expr.split():
        if token.lstrip('-').isdigit():
            output.append(token)   # operand → output
        elif token == '(':
            ops.append(token)
        elif token == ')':
            while ops and ops[-1] != '(':
                output.append(ops.pop())
            ops.pop()              # remove '('
        else:   # operator
            while (ops and ops[-1] != '(' and
                   precedence.get(ops[-1], 0) >= precedence[token]):
                output.append(ops.pop())
            ops.append(token)

    while ops: output.append(ops.pop())
    return " ".join(output)

print(infix_to_postfix("3 + 4 * 2"))          # "3 4 2 * +"
print(infix_to_postfix("( 3 + 4 ) * 2"))      # "3 4 + 2 *"

# ── Recursion → Iterative with explicit stack ────────
# Any recursive function CAN be converted to iterative
# using an explicit stack. Example: DFS on a tree/graph

# Recursive DFS (uses call stack implicitly):
def dfs_recursive(graph, node, visited=None):
    if visited is None: visited = set()
    visited.add(node)
    for neighbor in graph[node]:
        if neighbor not in visited:
            dfs_recursive(graph, neighbor, visited)
    return visited

# Iterative DFS (explicit stack — handles deep graphs):
def dfs_iterative(graph, start):
    visited = set()
    stack = [start]
    while stack:
        node = stack.pop()
        if node not in visited:
            visited.add(node)
            for neighbor in graph[node]:
                if neighbor not in visited:
                    stack.append(neighbor)
    return visited
